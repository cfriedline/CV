\section{Doctoral dissertation}
\cvline{Title}{\emph{Phylometagenomics: a new framework for uncovering microbial community diversity}}
\cvline{Committee}{Maria C. Rivera (advisor), Danail Bonchev, J. Paul Brooks, Rima B. Franklin,  
Donald R. Young}
\cvline{Description}{My analyses of the metagenomic 
sequence data from bacterial communities sampled from different environments demonstrate 
that the proper identification of the biological processes influencing these 
communities requires the development and implementation of new statistical and computational
methodologies that take advantage of the extensive amount of information
generated in next-generation, high-throughput sequencing projects.}

\section{Master's thesis}
\cvline{Title}{\emph{SLEDRIDE: Simplified Learning about Expression Data Running in a Desktop 
Environment}}
\cvline{Committee}{Paul Fawcett (advisor), James E. Ames, Catherine Dumur, Lemont B. Kier }
\cvline{Description}{SLEDRIDE is a tool that provides the ability to visually display and analyze 
microarray expression data in a variety of ways.  With this tool, it is possible to perform 
multi-experiment, genome-wide microarray studies and examine the correlation coefficients calculated 
from the dataset.  SLEDRIDE also makes it possible to view your data in genome order, which provides a 
very effective method  for identifying areas in a genome acting concordantly.}
\cvline{Website}{\url{http://sledride.csbc.vcu.edu}}
